equations{ 'inverter_num_units' } = define() {
	if( technology() == "Flat Plate PV" ) return ${inverter_count};
	elseif ( technology() == "PV_battery" ) return ${inverter_count};
	elseif ( technology() == "PVWatts" ) return 1;
	elseif ( technology() == "PVWatts_battery" ) return 1;
	elseif ( technology() == "Fuel Cell") return 1;
	else return 0;
};

equations{ 'module_power' } = define() {
	tech = technology();
	if ( tech == "PVWatts" || tech == "PVWatts_battery" || tech == "Fuel Cell") { 
		return ${system_capacity};
	}
	else {
		// Flat Plate PV
		return 0.001*?${module_model}
				[
					${spe_power},
					${cec_p_mp_ref},
					${6par_pmp},
					${snl_ref_pmp},
					${sd11par_Pmp0}
				];
	}
};

equations{ 'contingency' } = define() {
	return ${contingency_percent}/100*(
				 ${module_total}
				+${inverter_total}
				+${bos_equip_total}
				+${install_labor_total} 
				+${install_margin_total} );
};

equations{ 'inverter_power' } = define() {
	tech = technology();
	if ( tech == "PVWatts" || tech == "PVWatts_battery" || tech == "Fuel Cell") {
		return ${system_capacity}/${dc_ac_ratio};
	}
	else return 0.001*?${inverter_model}
			[
				${inv_snl_paco},
				${inv_ds_paco},
				${inv_pd_paco},
				${inv_cec_cg_paco}
			];
};

function show_battery_costs(t, f)
{
	show_battery = false;
	if (t == "PVWatts" &&((f=='Residential')||(f=='Commercial')||(f=='Third Party'))){
		show_battery = true;
	}
	else if (t == 'Flat Plate PV' && (f !='LCOE Calculator') && (f !='None')){
		show_battery = true;
	}
	else if (t == 'Fuel Cell'){
		show_battery = true;
	}
	
	return show_battery;

}
